        function startStage2AfterPopup() {

        function startStage2Game() {

            gameStarted = true;

            isGameOver = false;

            enemyHealth = maxEnemyHealth;

            document.getElementById("gameContainer").style.display = "block";

            updateEnemyHealthBar();

            Object.keys(playerHealth).forEach(updatePlayerHealthBar);

            startEnemyMotion();

            startEnemyAttacks();

        }

        function startEnemyMotion() {

            const enemy = document.getElementById("enemy-stage2");

            let y = 20;

            enemyInterval = setInterval(() => {

                if (isGameOver) return;

                y += enemyDirection * 5;

                if (y >= 80 || y <= -32) enemyDirection *= -1;

                enemy.style.top = y + "%";

            }, 70);

        }

        function startEnemyAttacks() {

            attackInterval = setInterval(() => {

                if (isGameOver || attackCooldown) return;

                const alive = Object.entries(playerHealth).filter(([id, v]) => v > 0);

                if (alive.length === 0) return loseGame();

                const [target] = alive[Math.floor(Math.random() * alive.length)];

                const name = target === "dan" ? "דן" : target === "gili" ? "גילי" : "יונתן";

                showWarning(`אוי לא! האיראני עומד לתקוף את ${name}!`);

                attackCooldown = true;

                const enemy = document.getElementById("enemy-stage2");

                setTimeout(() => enemy.style.backgroundImage = "url('https://raw.githubusercontent.com/Itay2802/dan_gili_yon/main/enemy_level2_mouthopen.png')", 1000);

                setTimeout(() => {

                    document.body.classList.add("shake");

                    enemy.style.backgroundImage = "url('https://raw.githubusercontent.com/Itay2802/dan_gili_yon/main/enemy_level2_tongue.png')";

                }, 1300);

                setTimeout(() => {

                    if (!document.getElementById(`${target}-stage2`).classList.contains("dodging")) {

                        hitPlayer(target);

                    }

                    enemy.style.backgroundImage = "url('https://raw.githubusercontent.com/Itay2802/dan_gili_yon/main/enemy_level2_mouthclosed.png')";

                    hideWarning();

                    document.body.classList.remove("shake");

                    attackCooldown = false;

                }, 1800);

            }, 3000);

        }

        function showWarning(msg) {

        function hideWarning() {

            document.getElementById("warningText").style.display = "none";

        function updateEnemyHealthBar() {

            const percent = Math.max(0, enemyHealth / maxEnemyHealth) * 100;

            document.getElementById("enemyHealthBar").style.width = percent + "%";

        }

        function updatePlayerHealthBar(id) {

            const health = playerHealth[id];

            const percent = Math.max(0, health / 4) * 100;

            document.getElementById(`${id}-health`).style.width = percent + "%";

            const el = document.getElementById(`${id}-stage2`);

            if (health <= 0) {

                el.style.opacity = "0.3";

                el.style.filter = "grayscale(1)";

            }

        }

        function hitPlayer(id) {

            if (!playerHealth[id]) return;

            playerHealth[id]--;

            updatePlayerHealthBar(id);

            if (playerHealth[id] <= 0) {

                const text = id === 'dan' ? 'דן הלך לשחק במחשב. אופסי' :

                    id === 'gili' ? 'גילי הלכה לצפות במשפחת ספיר. מצטערים' :

                    'יונתן ראה מישהו משחק במגנטים ועכשיו בא לו גם. הלך';

                showMessage(text);

            }

            if (Object.values(playerHealth).every(v => v <= 0)) loseGame();

        }

        function hitEnemy(amount) {

            if (isGameOver) return;

            enemyHealth -= amount;

            updateEnemyHealthBar();

            if (enemyHealth <= 0) winGame();

        }

        function showMessage(text) {

            const box = document.getElementById("messageBox");

            box.style.display = "block";

        }

        function closeMessage() {

            document.getElementById("messageBox").style.display = "none";

        }
